Create src/lib/schemas/blog.ts:
- Import z from 'zod'
- Define blogPostSchema with:
  - title: z.string().min(1)
  - description: z.string().min(1)
  - date: z.string().regex(/^\d{4}-\d{2}-\d{2}$/) for YYYY-MM-DD format
  - category: z.enum(['AI Automation', 'Travel Industry', 'Case Studies', 'Workflow Automation', 'Integrations'])
  - author: z.string().default('RomAIx Team')
  - readTime: z.number().positive().optional()
  - image: z.string().startsWith('/').or(z.string().url())
  - tags: z.array(z.string()).default([])
- Export type BlogPostMetadata = z.infer<typeof blogPostSchema>
